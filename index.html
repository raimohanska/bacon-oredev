var active = true;
var timeout = null;
function setInactive() {
  bubble("asleep?")
  active = false;
}
function setActive() {
  if (!active) {
    bubble("back!")
    active = true;
  }
  clearTimeout(timeout);
  timeout = setTimeout(setInactive, 10000);
}
$(window).on('keydown mousemove', setActive)

bubble("Hello")
